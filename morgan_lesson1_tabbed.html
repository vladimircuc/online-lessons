    // Read array from user

        cin >> arr[i];

    // Call the function

{
    double sum = 0;
    for(int i=0; i<size; i++)
    {
        arr[i] = arr[i] * arr[i];   // replace value with its square
        sum = sum + arr[i];          // add to running sum
    }
    return sum;
}

Pass by Value vs Pass by Reference

pass by VALUE — original unchanged
void triple(int x)
{
    x = x * 3;
    // x is a COPY — original not changed
}

int main()
{
    int num = 5;
    triple(num);
    cout << num;  // still 5!
}
pass by REFERENCE — original IS changed
void triple(int &x)   // & means reference
{
    x = x * 3;
    // x IS the original variable
}

int main()
{
    int num = 5;
    triple(num);
    cout << num;  // now 15!
}
reference_use_cases.cpp — when to use &
// Use & when: function needs to modify the original variable
void swap(int &a, int &b)
{
    int temp = a;
    a = b;
    b = temp;
    // swaps the real variables in main
}

// Use & when: returning more than one value
int calculate(int low, int high, int div, int &count)
{
    // count is returned by reference (second output)
    // function return value is the sum
    int sum = 0;
    count = 0;
    for(int i=low; i<=high; i++)
    {
        if (i % div == 0)
        {
            sum += i;
            count++;
        }
    }
    return sum;   // returns sum normally
    // count is "returned" through the reference parameter
}

int main()
{
    int totalCount;
    int totalSum = calculate(10, 40, 8, totalCount);
    // After call: totalSum = 72, totalCount = 4
}

Function Overloading & Default Parameters

overloading.cpp
// Same function name, different parameter signatures
// Compiler picks the right one based on what you pass

double calculate(double a, double b, char op = 'S')
{
    // Default parameter: if op not given, defaults to 'S'
    if (op == 'S') return a + b;   // sum
    if (op == 'D') return a - b;   // difference
    return 0;
}

int calculate(int low, int high, int div, int &count)
{
    // Different parameter types = different overload
    int sum = 0;
    count = 0;
    for(int i=low; i<=high; i++)
        if(i % div == 0) { sum += i; count++; }
    return sum;
}

int main()
{
    // These all call different versions:
    cout << calculate(3.0, 4.0);         // uses double version, op='S' (default)
    cout << calculate(3.0, 4.0, 'D');    // uses double version, op='D'

    int cnt;
    cout << calculate(10, 40, 8, cnt);   // uses int version
}

}

// Replace each element with sum of its left and right neighbors
// Position 0: left = -3, last position: right = 250
void sumReplace(int arr[], int size)
{
    int temp[10];
    for(int i=0; i<size; i++)
        temp[i] = arr[i];       // copy first — use original values

    for(int i=0; i<size; i++)
    {
        int left  = (i == 0)        ? -3  : temp[i-1];   // special left edge
        int right = (i == size-1)   ? 250 : temp[i+1];   // special right edge
        arr[i] = left + right;
    }
}

// Find and return the largest element — const means we won't modify it
int findMax(const int arr[], int size)
{
    int max = arr[0];               // assume first is max
    for(int i=1; i<size; i++)
    {
        if (arr[i] > max)
            max = arr[i];           // update if we find something bigger
    }
    return max;
}

🔍 Practice — Write the "moduli" function (from exam sheet)

15 ints. Function moduli: replace each element with arr[i] % arr[i+1]. If arr[i+1] == 0, leave it unchanged. Work from position 0.

moduli.cpp
#include<iostream>
using namespace std;

void moduli(int arr[], int size);

int main()
{
    int arr[15];
    cout << "Enter the array: ";
    for(int i=0; i<15; i++)
        cin >> arr[i];

    moduli(arr, 15);

    cout << "After moduli: ";
    for(int i=0; i<15; i++)
        cout << arr[i] << "\t";
    cout << endl;
    return 0;
}

void moduli(int arr[], int size)
{
    int temp[15];
    for(int i=0; i<size; i++)
        temp[i] = arr[i];          // copy first to avoid chain overwriting

    for(int i=0; i<size-1; i++)   // stop at size-1 (no "next" for last)
    {
        if (temp[i+1] == 0)
            arr[i] = temp[i];      // leave unchanged if divisor is 0
        else
            arr[i] = temp[i] % temp[i+1];
    }
    // last element: no next element to mod by, leave unchanged
}

#include<iostream>
#include<cstring>   // strlen, strcpy, strcat, strcmp
using namespace std;

int main()
{
    char str[100] = "Hello World";   // auto adds \0 at end
    char str2[100];

    // Length — does NOT count the \0 null terminator
    cout << strlen(str);        // 11

    // Copy — use strcpy, NOT str2 = str (that's a compile error)
    strcpy(str2, str);           // str2 now = "Hello World"

    // Concatenate — appends str2 to the end of str
    strcat(str, "!");            // str = "Hello World!"

    // Compare — 0 means equal, negative means str comes first alphabetically
    if (strcmp(str, str2) == 0)
        cout << "Same" << endl;

    // Read one word
    cin >> str;                  // stops at whitespace

    // Read full line
    cin.getline(str, 100);       // reads up to 99 chars + newline

    return 0;
}
